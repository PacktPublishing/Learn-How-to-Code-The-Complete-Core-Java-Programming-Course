package java_basics;

public class For_Loop {

	public static void main(String[] args) {
		
		for (int i = 1; i <= 100; i++) {
			if (i % 2 == 0) {
				System.out.println("Even Number:" + i);
			}
			
		}
		
		/**
		 * 
		 * Even Numbers 1 and 100
		 * 
		 * num % 2
		 * 
		 * 
		 */
		
		
		
		

	}

}
